## 1. Overview

### 1.1 Scope

| 필러 | 기능 | 신규/변경 |
|---|---|---|
| F1 | Form Builder — D&D UI, 필드 팔레트, 캔버스, 실시간 미리보기 | 신규 routes + 컴포넌트 |

| F4 | Conditional Fields — form_schema.conditionals, FormRenderer 조건부 렌더링 | 기존 스키마 확장 |

```sql
-- fn_resolve_approval_line: 기안자 + 폼 + 콘텐츠로 결재선 자동 생성
-- 반환: jsonb (ApprovalLineItem[]) — fn_submit_draft에 p_approval_line으로 전달
CREATE OR REPLACE FUNCTION public.fn_resolve_approval_line(
  p_tenant_id uuid,
  p_form_id uuid,
  p_content jsonb
)
RETURNS jsonb
LANGUAGE plpgsql
STABLE
SECURITY DEFINER
SET search_path = public
AS $$
DECLARE
  v_user_id uuid := auth.uid();
  v_rule record;
  v_cond jsonb;
  v_field_val text;
  v_cond_matched boolean;
  v_template jsonb;
  v_step jsonb;
  v_role text;
  v_step_cond jsonb;
  v_step_cond_matched boolean;
  v_result jsonb := '[]'::jsonb;
  v_drafter_dept_id uuid;
  v_dept_path text;
  v_approver_user_id uuid;
BEGIN
  -- 기안자 부서 경로 조회
  SELECT tm.department_id, d.path
    INTO v_drafter_dept_id, v_dept_path
  FROM public.tenant_members tm
  LEFT JOIN public.departments d ON d.id = tm.department_id
  WHERE tm.tenant_id = p_tenant_id AND tm.user_id = v_user_id;

  -- 매칭 규칙 탐색 (priority ASC, form_id 구체적일수록 우선)
  SELECT * INTO v_rule
  FROM public.approval_rules
  WHERE tenant_id = p_tenant_id
    AND is_active = true
    AND (form_id = p_form_id OR form_id IS NULL)
  ORDER BY
    (form_id IS NULL) ASC,   -- form_id 지정 규칙 우선
    priority ASC
  LIMIT 1;

  IF v_rule.id IS NULL THEN
    -- 매칭 규칙 없음 → 빈 배열 반환 (기안자가 수동 구성해야 함)
    RETURN '[]'::jsonb;
  END IF;

  -- condition_json 평가
  v_cond := v_rule.condition_json;
  IF v_cond IS NOT NULL THEN
    v_field_val := p_content ->> (v_cond->>'field');
    v_cond_matched := public.fn_eval_condition(v_cond, v_field_val);
    IF NOT v_cond_matched THEN
      RETURN '[]'::jsonb;
    END IF;
  END IF;

  -- line_template_json → ApprovalLineItem[] 변환
  FOR v_step IN SELECT * FROM jsonb_array_elements(v_rule.line_template_json) LOOP
    v_role := v_step->>'role';
    v_step_cond := v_step->'condition';

    -- 개별 step 조건 평가
    IF v_step_cond IS NOT NULL THEN
      v_field_val := p_content ->> (v_step_cond->>'field');
      v_step_cond_matched := public.fn_eval_condition(v_step_cond, v_field_val);
      IF NOT v_step_cond_matched THEN
        CONTINUE;
      END IF;
    END IF;

    -- role → user_id 해석
    v_approver_user_id := public.fn_resolve_role(
      p_tenant_id, v_user_id, v_drafter_dept_id, v_role
    );

    IF v_approver_user_id IS NULL THEN
      -- role 해석 실패 → 이 step 건너뜀 (R29 완화)
      CONTINUE;
    END IF;

    v_result := v_result || jsonb_build_object(
      'userId', v_approver_user_id,
      'stepType', COALESCE(v_step->>'stepType', 'approval')
    );
  END LOOP;

  RETURN v_result;
END;
$$;

GRANT EXECUTE ON FUNCTION public.fn_resolve_approval_line(uuid, uuid, jsonb)
  TO authenticated;


-- fn_eval_condition: 단일 조건 평가 헬퍼
CREATE OR REPLACE FUNCTION public.fn_eval_condition(
  p_cond jsonb,
  p_val text
)
RETURNS boolean
LANGUAGE plpgsql
STABLE
SECURITY DEFINER
SET search_path = public
AS $$
DECLARE
  v_op text := p_cond->>'operator';
  v_target text := p_cond->>'value';
BEGIN
  IF p_val IS NULL THEN RETURN false; END IF;
  RETURN CASE v_op
    WHEN 'eq'       THEN p_val = v_target
    WHEN 'neq'      THEN p_val <> v_target
    WHEN 'gte'      THEN p_val::numeric >= v_target::numeric
    WHEN 'lte'      THEN p_val::numeric <= v_target::numeric
    WHEN 'contains' THEN p_val ILIKE '%' || v_target || '%'
    ELSE false
  END;
EXCEPTION WHEN OTHERS THEN
  RETURN false;
END;
$$;


-- fn_resolve_role: role 문자열 → user_id
-- 조직도 상위 탐색 (departments.path LIKE 기반)
CREATE OR REPLACE FUNCTION public.fn_resolve_role(
  p_tenant_id uuid,
  p_drafter_id uuid,
  p_drafter_dept_id uuid,
  p_role text
)
RETURNS uuid
LANGUAGE plpgsql
STABLE
SECURITY DEFINER
SET search_path = public
AS $$
DECLARE
  v_user_id uuid;
  v_dept_path text;
  v_parent_path text;
  v_depth int;
BEGIN
  -- 직접 userId 지정 (JSON object role)
  IF p_role ~ '^[0-9a-f-]{36}$' THEN
    RETURN p_role::uuid;
  END IF;

  IF p_drafter_dept_id IS NULL THEN
    RETURN NULL;
  END IF;

  SELECT path INTO v_dept_path FROM public.departments
  WHERE id = p_drafter_dept_id;

  CASE p_role
    WHEN 'team_lead' THEN
      v_depth := 1;
    WHEN 'department_head' THEN
      v_depth := 2;
    WHEN 'ceo' THEN
      -- 최상위 부서의 멤버 중 level 최고 직급 사용자
      SELECT tm.user_id INTO v_user_id
      FROM public.tenant_members tm
      JOIN public.job_titles jt ON jt.id = tm.job_title_id
      WHERE tm.tenant_id = p_tenant_id
      ORDER BY jt.level DESC
      LIMIT 1;
      RETURN v_user_id;
    ELSE
      RETURN NULL;
  END CASE;

  -- path에서 v_depth 단계 상위 부서 계산
  -- path 예: '/hq/eng/platform' → 상위 1단계: '/hq/eng'
  v_parent_path := regexp_replace(
    v_dept_path,
    '(/[^/]+){' || v_depth || '}$',
    ''
  );

  IF v_parent_path = '' OR v_parent_path = v_dept_path THEN
    RETURN NULL;
  END IF;

  -- 상위 부서의 가장 높은 직급 멤버 선택
  SELECT tm.user_id INTO v_user_id
  FROM public.departments d
  JOIN public.tenant_members tm ON tm.department_id = d.id
    AND tm.tenant_id = p_tenant_id
  JOIN public.job_titles jt ON jt.id = tm.job_title_id
  WHERE d.tenant_id = p_tenant_id
    AND d.path = v_parent_path
  ORDER BY jt.level DESC
  LIMIT 1;

  RETURN v_user_id;
END;
$$;
```

### 4.5 Schema Generation

빌더 "발행" 클릭 시 `publishForm` action 호출:

```typescript
// builder/[formId]/+page.server.ts
export const actions: Actions = {
  saveFormDraft: async ({ request, locals, params }) => {
    // 권한 확인 후 is_published=false로 schema UPDATE
  },

  publishForm: async ({ request, locals, params }) => {
    if (!locals.user) return fail(401);
    const tenant = await resolveTenant(locals.supabase, locals.user.id, params.tenantSlug);
    if (!tenant || !['owner', 'admin'].includes(tenant.role)) return fail(403);

    const fd = await request.formData();
    const schemaRaw = fd.get('schema')?.toString() ?? '';
    let schema: unknown;
    try { schema = JSON.parse(schemaRaw); }
    catch { return fail(400, { error: 'JSON 파싱 실패' }); }

    // 현재 양식 조회
    const { data: current } = await locals.supabase
      .from('approval_forms')
      .select('id, version, is_published, parent_form_id')
      .eq('id', params.formId)
      .eq('tenant_id', tenant.id)
      .single();

    if (!current) return fail(404);

    if (current.is_published) {
      // 이미 발행된 양식 → 새 버전 INSERT
      const { data: newForm, error: err } = await locals.supabase
        .from('approval_forms')
        .insert({
          tenant_id: tenant.id,
          code: fd.get('code')?.toString(),
          name: fd.get('name')?.toString(),
          description: fd.get('description')?.toString() || null,
          schema,
          version: (current.version ?? 1) + 1,
          parent_form_id: current.parent_form_id ?? current.id,
          is_published: true,
        })
        .select('id')
        .single();
      if (err) return fail(400, { error: err.message });
      redirect(303, `/t/${params.tenantSlug}/admin/forms/builder/${newForm.id}`);
    } else {
      // 미발행 양식 → 그 자리에서 발행 (version 증가)
      const { error: err } = await locals.supabase
        .from('approval_forms')
        .update({ schema, is_published: true, version: (current.version ?? 1) + 1, updated_at: new Date().toISOString() })
        .eq('id', params.formId)
        .eq('tenant_id', tenant.id);
      if (err) return fail(400, { error: err.message });
      redirect(303, `/t/${params.tenantSlug}/admin/forms`);
    }
  }
};
```

```sql
SET search_path = public
```

## 7. Conditional Fields

### 7.1 Schema Structure

```json
{
  "version": 1,
  "fields": [
    { "id": "type", "type": "select", "label": "유형", "options": [{"value":"출장","label":"출장"},{"value":"재택","label":"재택"}] },
    { "id": "destination", "type": "text", "label": "목적지" },
    { "id": "office_location", "type": "text", "label": "근무 장소" }
  ],
  "conditionals": [
    {
      "if": { "fieldId": "type", "operator": "eq", "value": "출장" },
      "show": ["destination"],
      "hide": ["office_location"]
    },
    {
      "if": { "fieldId": "type", "operator": "eq", "value": "재택" },
      "show": ["office_location"],
      "hide": ["destination"]
    }
  ]
}
```

**규칙**:
- `conditionals`가 없으면 모든 필드 표시 (기존 동작)
- 같은 필드에 여러 규칙이 충돌하면 마지막 매칭 규칙 우선
- 기본 상태(조건 미충족): `show` 목록 필드는 숨김, `hide` 목록 필드는 표시 (toggle 의미)
- `conditionals`가 있을 때의 기본: 규칙에서 언급된 필드는 처음에 숨김, 조건 충족 시 표시

> 단순화 정책: `conditionals`에 명시된 필드 ID는 기본 숨김. `show` 규칙이 충족되면 표시. `hide` 규칙이 충족되면 강제 숨김.

### 7.2 FormRenderer Changes

```svelte
<!-- FormRenderer.svelte (수정) -->
<script lang="ts">
  import type { FormSchema, ConditionalRule } from '$lib/types/approval';
  import FormField from './FormField.svelte';

  type Props = {
    schema: FormSchema;
    tenantId: string;
    value: Record<string, unknown>;
    errors?: Record<string, string>;
    readonly?: boolean;
    onChange: (next: Record<string, unknown>) => void;
  };

  let { schema, tenantId, value, errors = {}, readonly = false, onChange }: Props = $props();

  // conditionals에서 참조된 field IDs 수집
  let conditionalFieldIds = $derived(
    new Set(
      (schema.conditionals ?? []).flatMap((c) => [...(c.show ?? []), ...(c.hide ?? [])])
    )
  );

  // 각 필드의 visible 상태 계산
  let visibleMap = $derived<Record<string, boolean>>(() => {
    const map: Record<string, boolean> = {};
    // 기본: conditionals에 언급된 필드는 숨김
    for (const id of conditionalFieldIds) map[id] = false;

    for (const rule of schema.conditionals ?? []) {
      const fieldVal = String(value[rule.if.fieldId] ?? '');
      const matched = evalCondition(rule.if.operator, fieldVal, String(rule.if.value));
      if (matched) {
        for (const id of rule.show ?? []) map[id] = true;
        for (const id of rule.hide ?? []) map[id] = false;
      }
    }
    return map;
  });

  function evalCondition(op: string, actual: string, target: string): boolean {
    switch (op) {
      case 'eq': return actual === target;
      case 'neq': return actual !== target;
      case 'gte': return Number(actual) >= Number(target);
      case 'lte': return Number(actual) <= Number(target);
      case 'contains': return actual.toLowerCase().includes(target.toLowerCase());
      default: return false;
    }
  }

  function updateField(id: string, next: unknown) {
    onChange({ ...value, [id]: next });
  }

  // submit 시 숨김 필드 값 제거 (호출자가 사용)
  export function getFilteredValue(): Record<string, unknown> {
    const filtered: Record<string, unknown> = {};
    for (const field of schema.fields) {
      if (visibleMap[field.id] !== false) {
        filtered[field.id] = value[field.id];
      }
    }
    return filtered;
  }
</script>

<div class="flex flex-col gap-4">
  {#each schema.fields as field (field.id)}
    {#if visibleMap[field.id] !== false}
      <FormField
        {field}
        {tenantId}
        value={value[field.id]}
        error={errors[field.id]}
        {readonly}
        onChange={(next) => updateField(field.id, next)}
      />
    {/if}
  {/each}
</div>
```

## 10. Error Handling

| 상황 | 처리 |
|---|---|
| fn_resolve_approval_line이 빈 배열 반환 | 기안 페이지에서 "결재선 자동 구성 실패. 수동으로 입력하세요." 안내 + 수동 모드 전환 |
| fn_resolve_role이 NULL 반환 (역할 해석 실패) | step 건너뜀 → 빈 배열 가능 → 위와 동일 |
| 부서 순환 참조 삽입 시도 | `fn_dept_path`에서 재귀 감지 → RAISE EXCEPTION → UI 오류 표시 |
| 양식 발행된 상태에서 재발행 실패 | 서버 INSERT 오류 → fail(400) + 메시지 표시 |
| D&D svelte-dnd-action 호환 불가 (R27) | native HTML5 Drag API fallback. `FieldCanvas`에서 `USE_DND_ACTION` 플래그로 분기 |
| conditionals 충돌 (같은 필드 show+hide 동시) | 마지막 매칭 규칙 우선 정책 적용 (v2.0 단순화) |
| FormRenderer getFilteredValue export | Svelte 5에서 컴포넌트 export 함수는 `$bindable` 또는 callback prop으로 대체. `onMount`에서 ref 전달 패턴 사용 |

## 11. Security Considerations

| 항목 | 대응 |
|---|---|
| approval_rules 직접 조회 | RLS: `is_tenant_member` SELECT, `is_tenant_admin` INSERT/UPDATE/DELETE |
| fn_resolve_approval_line 호출자 | SECURITY DEFINER, auth.uid() 기반 기안자 부서 조회 → 타 테넌트 데이터 접근 불가 |
| fn_resolve_role 조직도 탐색 | tenant_id 필터 전제로 쿼리 (타 테넌트 부서 참조 불가) |
| 양식 버전 INSERT (새 버전) | RLS `af_admin_insert`: tenant_id NOT NULL AND is_tenant_admin. 관리자만 가능 |
| 조건부 필드 서버 우회 | 숨김 필드 값도 p_content에 포함될 수 있음. fn_submit_draft는 content를 그대로 저장 → 클라이언트 필터링은 UX 목적. 서버에서 특정 필드 strip은 v2.0 scope 외 (form_schema로 검증 시 추가 가능) |
| 부서 트리 depth 제한 | fn_dept_path 재귀 깊이 제한: `CYCLE` 감지 or depth 카운터 (R28). 권고: `MAX_DEPTH = 5` |
| auth.uid() InitPlan 규칙 | 모든 신규 RLS 정책에서 `(SELECT auth.uid())` 래핑 적용 |

## 부록 B — svelte-dnd-action Svelte 5 대체 방안 (R27)

svelte-dnd-action가 Svelte 5 runes와 호환되지 않을 경우:

```svelte
<!-- FieldCanvas.svelte — native HTML5 Drag API fallback -->
<script lang="ts">
  let dragIndex = $state<number | null>(null);

  function onDragStart(index: number) {
    dragIndex = index;
  }

  function onDrop(targetIndex: number) {
    if (dragIndex === null) return;
    const updated = [...fields];
    const [moved] = updated.splice(dragIndex, 1);
    updated.splice(targetIndex, 0, moved);
    onFieldsChange(updated);
    dragIndex = null;
  }
</script>

{#each fields as field, i (field.id)}
  <div
    draggable="true"
    ondragstart={() => onDragStart(i)}
    ondragover={(e) => e.preventDefault()}
    ondrop={() => onDrop(i)}
    class:opacity-50={dragIndex === i}
  >
    <CanvasFieldItem ... />
  </div>
{/each}
```
