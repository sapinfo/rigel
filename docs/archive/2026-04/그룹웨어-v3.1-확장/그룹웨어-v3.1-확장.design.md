CREATE TABLE public.board_posts (

  board_id uuid NOT NULL REFERENCES public.boards(id) ON DELETE CASCADE,
  tenant_id uuid NOT NULL REFERENCES public.tenants(id) ON DELETE CASCADE,
  title text NOT NULL,
  content jsonb NOT NULL DEFAULT '{}',   -- Tiptap JSON (RichEditor 재사용)
  author_id uuid NOT NULL REFERENCES auth.users(id),
  is_pinned boolean NOT NULL DEFAULT false,
  view_count integer NOT NULL DEFAULT 0,
  created_at timestamptz NOT NULL DEFAULT now(),
  updated_at timestamptz NOT NULL DEFAULT now()
);

CREATE INDEX board_posts_board_idx ON public.board_posts (board_id, created_at DESC);
CREATE INDEX board_posts_tenant_idx ON public.board_posts (tenant_id);

ALTER TABLE public.board_posts ENABLE ROW LEVEL SECURITY;

-- SELECT: 해당 board 접근 가능한 멤버 (boards RLS 통과 = 접근 가능)
-- boards와 양방향 EXISTS 방지: SECURITY DEFINER 헬퍼 사용
CREATE OR REPLACE FUNCTION public.can_access_board(p_board_id uuid)
  RETURNS boolean
  LANGUAGE sql STABLE SECURITY DEFINER
  SET search_path = public
AS $$
  SELECT EXISTS (
    SELECT 1 FROM public.boards b
    WHERE b.id = p_board_id
    AND public.is_tenant_member(b.tenant_id)
    AND (
      b.board_type = 'general'
      OR b.department_id IN (
        SELECT ud.department_id FROM public.user_departments ud
        WHERE ud.user_id = auth.uid() AND ud.tenant_id = b.tenant_id
      )
    )
  );
$$;

CREATE POLICY board_posts_select ON public.board_posts
  FOR SELECT TO authenticated
  USING (public.can_access_board(board_id));

CREATE POLICY board_posts_insert ON public.board_posts
  FOR INSERT TO authenticated
  WITH CHECK (
    public.can_access_board(board_id)
    AND author_id = (SELECT auth.uid())
  );

-- UPDATE/DELETE: 작성자 OR admin
CREATE POLICY board_posts_update ON public.board_posts
  FOR UPDATE TO authenticated
  USING (
    author_id = (SELECT auth.uid())
    OR public.is_tenant_admin(tenant_id)
  );

CREATE POLICY board_posts_delete ON public.board_posts
  FOR DELETE TO authenticated
  USING (
    author_id = (SELECT auth.uid())
    OR public.is_tenant_admin(tenant_id)
  );

-- 조회수 증가 RPC (SELECT FOR UPDATE 없이, 단순 increment)
CREATE OR REPLACE FUNCTION public.increment_post_view(p_post_id uuid)
  RETURNS void
  LANGUAGE sql SECURITY DEFINER
  SET search_path = public
AS $$
  UPDATE public.board_posts
  SET view_count = view_count + 1
  WHERE id = p_post_id;
$$;

-- 미확인 공지 수 RPC
CREATE OR REPLACE FUNCTION public.get_unread_announcement_count(p_tenant_id uuid)
  RETURNS integer
  LANGUAGE sql STABLE SECURITY DEFINER
  SET search_path = public
AS $$
  SELECT count(*)::integer
  FROM public.announcements a
  WHERE a.tenant_id = p_tenant_id
    AND a.published_at IS NOT NULL
    AND (a.expires_at IS NULL OR a.expires_at > now())
    AND a.require_read_confirm = true
    AND NOT EXISTS (
      SELECT 1 FROM public.announcement_reads ar
      WHERE ar.announcement_id = a.id AND ar.user_id = auth.uid()
    );
$$;

-- 팝업 공지 조회 RPC (로그인 시 호출)
CREATE OR REPLACE FUNCTION public.get_popup_announcements(p_tenant_id uuid)
  RETURNS SETOF public.announcements
  LANGUAGE sql STABLE SECURITY DEFINER
  SET search_path = public
AS $$
  SELECT a.*
  FROM public.announcements a
  WHERE a.tenant_id = p_tenant_id
    AND a.is_popup = true
    AND a.published_at IS NOT NULL
    AND (a.expires_at IS NULL OR a.expires_at > now())
    AND NOT EXISTS (
      SELECT 1 FROM public.announcement_reads ar
      WHERE ar.announcement_id = a.id AND ar.user_id = auth.uid()
    )
  ORDER BY a.published_at DESC;
$$;
```

CREATE TABLE public.attendance_records (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id uuid NOT NULL REFERENCES public.tenants(id) ON DELETE CASCADE,
  user_id uuid NOT NULL REFERENCES auth.users(id),

  clock_out timestamptz,
  work_type text NOT NULL DEFAULT 'normal'

  (tenant_id, work_date DESC);

  (user_id, work_date DESC);

ALTER TABLE public.attendance_records ENABLE ROW LEVEL SECURITY;

-- SELECT: 본인 기록 + admin은 전체
CREATE POLICY attendance_records_select ON public.attendance_records
  FOR SELECT TO authenticated
  USING (
    public.is_tenant_member(tenant_id)
    AND (
      user_id = (SELECT auth.uid())
      OR public.is_tenant_admin(tenant_id)
    )
  );

-- INSERT: 본인만 (오늘 날짜만)
CREATE POLICY attendance_records_insert ON public.attendance_records
  FOR INSERT TO authenticated
  WITH CHECK (
    user_id = (SELECT auth.uid())
    AND public.is_tenant_member(tenant_id)
  );

-- UPDATE: 본인 (clock_out만) OR admin
CREATE POLICY attendance_records_update ON public.attendance_records
  FOR UPDATE TO authenticated
  USING (
    user_id = (SELECT auth.uid())
    OR public.is_tenant_admin(tenant_id)
  );

-- 출근 RPC: idempotent (이미 있으면 무시)
CREATE OR REPLACE FUNCTION public.fn_clock_in(p_tenant_id uuid)
  RETURNS public.attendance_records
  LANGUAGE plpgsql SECURITY DEFINER
  SET search_path = public
AS $$
DECLARE
  v_today date := (now() AT TIME ZONE 'Asia/Seoul')::date;
  v_now timestamptz := now();
  v_settings attendance_settings;
  v_type text := 'normal';
  v_rec attendance_records;
BEGIN
  -- 근무 설정 조회
  SELECT * INTO v_settings FROM attendance_settings WHERE tenant_id = p_tenant_id;

  -- 지각 판정
  IF v_settings IS NOT NULL THEN
    IF (v_now AT TIME ZONE 'Asia/Seoul')::time >
       (v_settings.work_start_time + (v_settings.late_threshold_minutes || ' minutes')::interval) THEN
      v_type := 'late';
    END IF;
  END IF;

  INSERT INTO attendance_records (tenant_id, user_id, work_date, clock_in, work_type)
  VALUES (p_tenant_id, auth.uid(), v_today, v_now, v_type)
  ON CONFLICT (tenant_id, user_id, work_date) DO NOTHING
  RETURNING * INTO v_rec;

  -- 이미 출근한 경우 기존 레코드 반환
  IF v_rec IS NULL THEN
    SELECT * INTO v_rec FROM attendance_records
    WHERE tenant_id = p_tenant_id AND user_id = auth.uid() AND work_date = v_today;
  END IF;

  RETURN v_rec;
END;
$$;

-- 퇴근 RPC
CREATE OR REPLACE FUNCTION public.fn_clock_out(p_tenant_id uuid)
  RETURNS public.attendance_records
  LANGUAGE plpgsql SECURITY DEFINER
  SET search_path = public
AS $$
DECLARE
  v_today date := (now() AT TIME ZONE 'Asia/Seoul')::date;
  v_rec attendance_records;
BEGIN
  UPDATE attendance_records
  SET clock_out = now()
  WHERE tenant_id = p_tenant_id AND user_id = auth.uid() AND work_date = v_today
  RETURNING * INTO v_rec;

  IF v_rec IS NULL THEN
    RAISE EXCEPTION 'No clock-in record found for today';
  END IF;

  RETURN v_rec;
END;
$$;
```

## 8. 리스크 & 완화

| # | 리스크 | 영향 | 완화 |
|---|---|---|---|
| R1 | `can_access_board` 헬퍼 성능 | 게시글 조회 시 board JOIN | SECURITY DEFINER로 RLS bypass, 인덱스 활용 |
| R2 | `btree_gist` 확장 누락 | EXCLUDE constraint 실패 | 0061에서 `CREATE EXTENSION IF NOT EXISTS btree_gist` 선행 |
| R3 | 사이드바 전환 regression | 기존 결재 흐름 UX 변경 | 헤더 → 사이드바 점진 전환, 모바일은 기존 패턴 유지 |
| R4 | Tiptap content 크기 | jsonb 저장 용량 | 이미지는 Storage에 업로드, content에는 URL만 저장 |
| R5 | 초과근무 양식 seed 중복 | 기존 테넌트에 overtime form 존재 시 | `ON CONFLICT` 또는 `WHERE NOT EXISTS` 사용 |
| R6 | 팝업 공지 남용 | UX 피로 | expires_at 필수 권장, 최대 3개 동시 팝업 제한 (클라이언트) |
| R7 | 회의실 EXCLUDE constraint + RLS 조합 | 동시 예약 race | SECURITY DEFINER RPC로 예약 처리 (RLS 우회 + 제약 검증) |
